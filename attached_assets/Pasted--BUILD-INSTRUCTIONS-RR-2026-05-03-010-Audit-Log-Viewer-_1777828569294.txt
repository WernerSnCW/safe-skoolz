### GET /audit

```typescript
router.get(
  "/audit",
  authMiddleware,
  requireRole("coordinator", "head_teacher"),
  async (req, res) => {
    const user = (req as any).user as JwtPayload;

    // Parse query params
    const eventType = req.query.eventType as string | undefined;
    const actorRole = req.query.actorRole as string | undefined;
    const from     = req.query.from as string | undefined;
    const to       = req.query.to   as string | undefined;
    const cursor   = req.query.cursor as string | undefined;
    const limit    = Math.min(Number(req.query.limit ?? 50), 200);

    // Build WHERE conditions — always scope to schoolId
    const conditions = [eq(auditLogTable.schoolId, user.schoolId)];
    if (eventType) conditions.push(eq(auditLogTable.eventType, eventType));
    if (actorRole) conditions.push(eq(auditLogTable.actorRole, actorRole));
    if (from) conditions.push(gte(auditLogTable.createdAt, new Date(from)));
    if (to)   conditions.push(lte(auditLogTable.createdAt, new Date(to)));

    // Keyset cursor: base64-encoded JSON { createdAt, id }
    if (cursor) {
      const { createdAt, id } = JSON.parse(
        Buffer.from(cursor, "base64").toString()
      );
      conditions.push(
        or(
          lt(auditLogTable.createdAt, new Date(createdAt)),
          and(
            eq(auditLogTable.createdAt, new Date(createdAt)),
            lt(auditLogTable.id, id)
          )
        )
      );
    }

    const rows = await db
      .select()
      .from(auditLogTable)
      .where(and(...conditions))
      .orderBy(desc(auditLogTable.createdAt), desc(auditLogTable.id))
      .limit(limit + 1); // fetch one extra to detect next page

    const hasMore = rows.length > limit;
    const data = hasMore ? rows.slice(0, limit) : rows;

    const nextCursor = hasMore
      ? Buffer.from(
          JSON.stringify({
            createdAt: data[data.length - 1].createdAt,
            id: data[data.length - 1].id,
          })
        ).toString("base64")
      : null;

    res.json({ data, nextCursor, hasMore });
  }
);
```

### Data fetching

```typescript
const [eventTypeFilter, setEventTypeFilter] = useState("");
const [cursor, setCursor] = useState<string | null>(null);
const [allRows, setAllRows] = useState<AuditRow[]>([]);

// Fetch current page
const { data, isLoading } = useQuery({
  queryKey: ["/api/audit", eventTypeFilter, cursor],
  queryFn: async () => {
    const params = new URLSearchParams({ limit: "50" });
    if (eventTypeFilter) params.set("eventType", eventTypeFilter);
    if (cursor) params.set("cursor", cursor);
    const token = localStorage.getItem("safeschool_token");
    const res = await fetch(`${apiBase}/api/audit?${params}`, {
      headers: { Authorization: `Bearer ${token}` },
    });
    return res.json();
  },
});

// Accumulate rows for infinite scroll / load more
useEffect(() => {
  if (data?.data) {
    if (!cursor) {
      setAllRows(data.data); // fresh filter — reset list
    } else {
      setAllRows((prev) => [...prev, ...data.data]);
    }
  }
}, [data]);
```

### Page structure

```
Page heading: "Audit Log"
Subheading: "Complete record of all actions in this school"

[Filter bar]
  Event type dropdown: "All events" + each of the 41 event types
  (fetch from GET /audit/event-types on mount)
  Clear filters button (when filter active)

[Audit table]
  Columns: Date/Time | Event | Actor | Target | IP Address
  Rows: one per audit log entry
  Date/Time: formatted "DD MMM YYYY HH:mm"
  Event: eventType displayed as readable label
    (e.g. "incident_created" → "Incident created")
    Use a simple replace: eventType.replace(/_/g, " ")
  Actor: actorRole + actorId (show first 8 chars of UUID, or "System"
    if actorId is null)
  Target: targetType + targetId (first 8 chars, or "—" if null)
  IP Address: ipAddress or "—"

[Load more button]
  Visible when hasMore === true
  On click: setCursor(data.nextCursor)
  Label: "Load more"
  Shows spinner while loading

[Empty state]
  When no rows: "No audit log entries found."
  When filter active and no rows: "No entries match this filter."
```
